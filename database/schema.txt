4.cases
CREATE TABLE cases (
    case_id SERIAL PRIMARY KEY,
    client_id INT NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    advocate_id INT REFERENCES users(user_id) ON DELETE SET NULL,

    case_title VARCHAR(200) NOT NULL,
    case_type VARCHAR(50),
    case_description TEXT,

    status VARCHAR(20) DEFAULT 'PENDING'
        CHECK (status IN ('PENDING', 'APPROVED', 'REJECTED', 'ONGOING', 'CLOSED')),

    next_hearing_date DATE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

11.CREATE OR REPLACE FUNCTION handle_admin_review()
RETURNS TRIGGER
AS $$
BEGIN
    UPDATE cases
    SET status = NEW.decision
    WHERE case_id = NEW.case_id;

    INSERT INTO notifications (user_id, message)
    VALUES (
        (SELECT client_id FROM cases WHERE case_id = NEW.case_id),
        'Your case has been ' || NEW.decision
    );

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

12.CREATE TRIGGER trg_admin_review
AFTER INSERT ON admin_case_review
FOR EACH ROW
EXECUTE FUNCTION handle_admin_review();

13.CREATE OR REPLACE FUNCTION handle_case_assignment()
RETURNS TRIGGER
AS $$
BEGIN
    UPDATE cases
    SET advocate_id = NEW.advocate_id,
        status = 'ONGOING'
    WHERE case_id = NEW.case_id;

    INSERT INTO notifications (user_id, message)
    VALUES (
        NEW.advocate_id,
        'A new case has been assigned to you.'
    );

    INSERT INTO notifications (user_id, message)
    VALUES (
        (SELECT client_id FROM cases WHERE case_id = NEW.case_id),
        'An advocate has been assigned to your case.'
    );

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

14.CREATE TRIGGER trg_case_assignment
AFTER INSERT ON case_assignments
FOR EACH ROW
EXECUTE FUNCTION handle_case_assignment();
